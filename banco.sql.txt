create table Clientes(
       id_cliente serial,
	   nome varchar,
	   idade int,
	   
	   constraint pk_cliente primary key(id_cliente)
	 

);
create table Endereco(
	    id_cliente int,
        rua varchar,
	    cidade varchar,
	    bairro varchar,
		constraint cliente_fk foreign key (id_cliente) references Clientes(id_cliente)
);
